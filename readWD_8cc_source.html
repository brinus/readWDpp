274DAQEvent &DAQEvent::TimeCalibration(const unsigned short &tCell, const std::vector<float> &times, int i, int j)
275{
276 times_[i][j] = times;

289DAQEvent &DAQEvent::EvalPedestal()
290{
291 if (!is_getch_)
292 {
293 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
294 exit(0);
295 }
296
297 int ped_interval_dist = ped_interval_.second - ped_interval_.first;
298 const vector<float> &volts = volts_[ch_.first][ch_.second];
299 ped_ = {0., 0.};
300 ped_.first = accumulate(volts.begin() + ped_interval_.first, volts.begin() + ped_interval_.second, 0.) / ped_interval_dist;
301 for (int i = ped_interval_.first; i < ped_interval_.second; ++i)
302 {
303 ped_.second += pow(volts[i] - ped_.first, 2);
304 }
305 ped_.second = sqrt(ped_.second / ped_interval_dist);
306
307 return *this;
308}

315DAQEvent &DAQEvent::EvalIntegrationBounds()
316{
317 if (!is_getch_)
318 {
319 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
320 exit(0);
321 }
322
323 iw_ = {indexMin_[0], indexMin_[0]};
324
325 const vector<float> &volts = volts_[ch_.first][ch_.second];
326 auto lower_bound = ped_.first - 5 * ped_.second;
327
328 if (peak_.first < lower_bound)
329 {
330 while (volts[iw_.first] < lower_bound and iw_.first > 0)
331 {
332 --iw_.first;
333 }
334
335 while (volts[iw_.second] < lower_bound and iw_.second < SAMPLES_PER_WAVEFORM - 1)
336 {
337 ++iw_.second;
338 }
339 }
340
341 return *this;

349DAQEvent &DAQEvent::FindPeaks()
350{
351 if (!is_getch_)
352 {
353 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
354 exit(0);
355 }
356
357 const vector<float> &volts = volts_[ch_.first][ch_.second];
358 const vector<float> &times = times_[ch_.first][ch_.second];
359 indexMin_ = {};
360 auto index_min = distance(volts.begin(), min_element(volts.begin(), volts.end()));
361
362 for (int i = 1; i < SAMPLES_PER_WAVEFORM - 1; ++i)
363 {
364 auto signal = volts[i] - ped_.first < - 5 * ped_.second;
365 auto min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
366 auto min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
367 auto at_least = volts[i] < volts[index_min] * 0.5;
368 if (signal and min_left and min_right and at_least)
369 {
370 indexMin_.push_back(i);
371 }
372 }
373
374 if (find(indexMin_.begin(), indexMin_.end(), index_min) == indexMin_.end())
375 {
376 auto pos = find_if(indexMin_.begin(), indexMin_.end(), [index_min](auto i){ return i > index_min;});
377 indexMin_.insert(pos, index_min);
378 }
379
380 else if (indexMin_.size() == 0)
381 {
382 indexMin_.push_back(index_min);
383 }
384
385 peak_ = {volts[indexMin_[0]], times[indexMin_[0]]};
386
387 return *this;
388}

714DAQFile::operator bool()
715{
716 map<char, char> header{{'E', 'B'}, {'B', 'C'}, {'C', 'B'}};
717 if (!in_.good())
718 {
719 cout << "End of file reached" << endl;
720 return 0;
721 }
722 else if (n_ == 'T' or n_ == 'D') // Ignores DRSx and TIME
723 {
724 return 0;
725 }
726 else if (n_ == o_) // C --> C, B --> B
727 return 1;
728 else if (n_ == header[o_]) // E --> B, B --> C, C --> B
729 {
730 o_ = n_;
731 return 1;
732 }
733 else if (n_ == 'E' and initialization_ == 0) // End of initialization, first EHDR
734 {
735 cout << "Initialization done --> EHDR next" << endl;
736 initialization_ = 1;
737 return 0;
738 }
739 return 0;
