465DAQEvent &DAQEvent::EvalPedestal()
466{
467 if (!is_getch_)
468 {
469 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
470 exit(0);
471 }
472
473 int ped_interval_dist = ped_interval_.second - ped_interval_.first;
474 const vector<float> &volts = volts_[ch_.first][ch_.second];
475 ped_ = {0., 0.};
476 ped_.first = accumulate(volts.begin() + ped_interval_.first, volts.begin() + ped_interval_.second, 0.) / ped_interval_dist;
477 for (int i = ped_interval_.first; i < ped_interval_.second; ++i)
478 {
479 ped_.second += pow(volts[i] - ped_.first, 2);
480 }
481 ped_.second = sqrt(ped_.second / ped_interval_dist);
482
483 return *this;

530DAQEvent &DAQEvent::FindPeaks()
531{
532 if (!is_getch_)
533 {
534 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
535 exit(0);
536 }
537
538 auto &volts = volts_[ch_.first][ch_.second];
539 auto &times = times_[ch_.first][ch_.second];
540 indexMin_ = {};
541
542 if (user_iw_) // IW set by the user
543 {
544 auto index_min = distance(volts.begin(), min_element(volts.begin() + iw_.first, volts.begin() + iw_.second));
545 indexMin_.push_back(index_min);
546 }
547 else if (peak_threshold_ == 1.) // No threshold level set by the user
548 {
549 auto index_min = distance(volts.begin(), min_element(volts.begin(), volts.end()));
550 for (int i = 1; i < SAMPLES_PER_WAVEFORM - 1; ++i)
551 {
552 auto signal = volts[i] - ped_.first < -5 * ped_.second;
553 auto min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
554 auto min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
555 auto at_least = volts[i] < volts[index_min] * 0.5;
556 if (signal and min_left and min_right and at_least)
557 {
558 indexMin_.push_back(i);
559 }
560 }
561
562 if (find(indexMin_.begin(), indexMin_.end(), index_min) == indexMin_.end()) // Sorted insertion of min element's index
563 {
564 auto pos = find_if(indexMin_.begin(), indexMin_.end(), [index_min](auto i)
565 { return i > index_min; });
566 indexMin_.insert(pos, index_min);
567 }
568 }
569 else // Threshold level set by the user
570 {
571 for (int i = 1; i < SAMPLES_PER_WAVEFORM - 1; ++i)
572 {
573 auto signal = volts[i] - ped_.first < -5 * ped_.second;
574 auto min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
575 auto min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
576 auto at_least = volts[i] < peak_threshold_;
577 if (signal and (min_left or min_right) and at_least)
578 {
579 indexMin_.push_back(i);
580 }
581 }
582 }
583
584 if (indexMin_.size() == 0)
585 {
586 auto index_min = distance(volts.begin(), min_element(volts.begin(), volts.end()));
587 indexMin_.push_back(index_min);
588 }
589
590 peak_ = {volts[indexMin_[0]], times[indexMin_[0]]};
591
592 return *this;
593}
