279{

438DAQEvent &DAQEvent::EvalPedestal()
439{
440 if (!is_getch_)
441 {
442 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
443 exit(0);
444 }
445
446 int ped_interval_dist = ped_interval_.second - ped_interval_.first;
447 const vector<float> &volts = volts_[ch_.first][ch_.second];
448 ped_ = {0., 0.};
449 ped_.first = accumulate(volts.begin() + ped_interval_.first, volts.begin() + ped_interval_.second, 0.) / ped_interval_dist;
450 for (int i = ped_interval_.first; i < ped_interval_.second; ++i)
451 {
452 ped_.second += pow(volts[i] - ped_.first, 2);
453 }
454 ped_.second = sqrt(ped_.second / ped_interval_dist);
455
456 return *this;
457}

464DAQEvent &DAQEvent::EvalIntegrationBounds()
465{
466 if (!is_getch_)
467 {
468 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
469 exit(0);
470 }
471
472 if (user_iw_)
473 {
474 return *this;

476
477 iw_ = {indexMin_[0], indexMin_[0]};
478

480 auto lower_bound = ped_.first - 5 * ped_.second;

503DAQEvent &DAQEvent::FindPeaks()
504{
505 if (!is_getch_)
506 {
507 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
508 exit(0);
509 }
510
511 auto &volts = volts_[ch_.first][ch_.second];
512 auto &times = times_[ch_.first][ch_.second];
513 indexMin_ = {};
514
515 if (peak_threshold_ == 1.) // No threshold level set by the user
516 {
517 auto index_min = distance(volts.begin(), min_element(volts.begin(), volts.end()));
518 for (int i = 1; i < SAMPLES_PER_WAVEFORM - 1; ++i)
519 {
520 auto signal = volts[i] - ped_.first < -5 * ped_.second;
521 auto min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
522 auto min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
523 auto at_least = volts[i] < volts[index_min] * 0.5;
524 if (signal and min_left and min_right and at_least)
525 {
526 indexMin_.push_back(i);
527 }
528 }
529
530 if (find(indexMin_.begin(), indexMin_.end(), index_min) == indexMin_.end()) // Sorted insertion of min element's index
531 {
532 auto pos = find_if(indexMin_.begin(), indexMin_.end(), [index_min](auto i)
533 { return i > index_min; });
534 indexMin_.insert(pos, index_min);
535 }
536 }
537 else // Threshold level set by the user
538 {
539 if (user_iw_) // IW set by the user
540 {
541 auto index_min = distance(volts.begin(), min_element(volts.begin() + iw_.first, volts.begin() + iw_.second));
542 indexMin_.push_back(index_min);
543 }
544 else // IW not set by the user
545 {
546 for (int i = 1; i < SAMPLES_PER_WAVEFORM - 1; ++i)
547 {
548 auto signal = volts[i] - ped_.first < -5 * ped_.second;
549 auto min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
550 auto min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
551 auto at_least = volts[i] < peak_threshold_;
552 if (signal and (min_left or min_right) and at_least)
553 {
554 indexMin_.push_back(i);
555 }
556 }
557 }
558 }
559
560 if (indexMin_.size() == 0)
561 {
562 auto index_min = distance(volts.begin(), min_element(volts.begin(), volts.end()));
563 indexMin_.push_back(index_min);
564 }
565
566 peak_ = {volts[indexMin_[0]], times[indexMin_[0]]};
567
568 return *this;
569}
