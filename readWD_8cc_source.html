521DAQEvent &DAQEvent::EvalPedestal()
522{
523 if (!is_getch_)
524 {
525 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
526 exit(0);
527 }
528
529 ped_interval_ = config_.pedInterval_[ch_.first][ch_.second];
530 int ped_interval_dist = ped_interval_.second - ped_interval_.first;
531 const vector<float> &volts = volts_[ch_.first][ch_.second];
532 ped_ = {0., 0.};
533 ped_.first = accumulate(volts.begin() + ped_interval_.first, volts.begin() + ped_interval_.second, 0.) / ped_interval_dist;
534 for (int i = ped_interval_.first; i < ped_interval_.second; ++i)
535 {
536 ped_.second += pow(volts[i] - ped_.first, 2);
537 }
538 ped_.second = sqrt(ped_.second / ped_interval_dist);
539
540 return *this;

548DAQEvent &DAQEvent::EvalIntegrationBounds()
549{
550 if (!is_getch_)
551 {
552 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
553 exit(0);
554 }
555
556 if (config_.user_iw_[ch_.first][ch_.second])
557 {
558 return *this;
559 }
560
561 iw_ = {indexMin_[0], indexMin_[0]};
562
563 const vector<float> &volts = volts_[ch_.first][ch_.second];
564 auto lower_bound = ped_.first - 5 * ped_.second;
565
566 if (peak_.first < lower_bound)
567 {
568 while (volts[iw_.first] < lower_bound and iw_.first > 0)
569 {
570 --iw_.first;
571 }
572
573 while (volts[iw_.second] < lower_bound and iw_.second < SAMPLES_PER_WAVEFORM - 1)
574 {
575 ++iw_.second;
576 }
577 }
578
579 config_.intWindow_[ch_.first][ch_.second] = {iw_.first, iw_.second};

588DAQEvent &DAQEvent::FindPeaks()
589{
590 if (!is_getch_)
591 {
592 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
593 exit(0);
594 }
595
596 long index_min;
597 auto &volts = volts_[ch_.first][ch_.second];
598 auto &times = times_[ch_.first][ch_.second];
599 indexMin_ = {};
600
601 iw_ = config_.intWindow_[ch_.first][ch_.second];
602 peak_threshold_ = config_.peakThr_[ch_.first][ch_.second];
603
604 if (config_.user_iw_[ch_.first][ch_.second]) // Integration window set by the user
605 {
606 index_min = distance(volts.begin(), min_element(volts.begin() + iw_.first, volts.begin() + iw_.second));
607 indexMin_.push_back(index_min);
608 }
609 else // No user integration window set
610 {
611 index_min = distance(volts.begin() + 2, min_element(volts.begin() + 2, volts.end() - 2)) + 2;
612 bool signal, min_left, min_rigth, at_lest;
613 for (int i = 2; i < SAMPLES_PER_WAVEFORM - 2; ++i)
614 {
615 signal = abs(volts[i] - ped_.first) > 5 * ped_.second;
616 min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
617 min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
618 if (config_.peakThr_[ch_.first][ch_.second] == 0.5) // Default threshold level
619 {
620 at_least = abs(volts[i] - ped_.first) > abs(volts[index_min] - ped_.first) * 0.5;
621 }
622 else // Custom threshold level
623 {
624 at_least = volts[i] < peak_threshold_;
625 }
626
627 if (signal and min_left and min_right and at_least)
628 {
629 indexMin_.push_back(i);
630 }
631 }
632
633 if (find(indexMin_.begin(), indexMin_.end(), index_min) == indexMin_.end()) // Sorted insertion of min element's index
634 {
635 auto pos = find_if(indexMin_.begin(), indexMin_.end(), [index_min](auto i)
636 { return i > index_min; });
637 indexMin_.insert(pos, index_min);
638 }
639 }
640
641 if (indexMin_.size() == 0) // Assure that at least global minimum is inserted in indexMin_
642 {
643 index_min = distance(volts.begin() + 2, min_element(volts.begin() + 2, volts.end() - 2)) + 2;
644 indexMin_.push_back(index_min);
645 }
646
647 peak_ = {volts[indexMin_[0]], times[indexMin_[0]]}; // First local minimum taken as peak
648
649 return *this;
650}

1301DAQFile::operator bool()
1302{

1304 if (!in_.good())
1305 {
1306 cout << "End of file reached" << endl;
1307 return 0;
1308 }
1309 else if (n_ == 'T' or n_ == 'D') // Ignores DRSx and TIME
1310 return 0;
1311 else if (n_ == o_) // C --> C, B --> B
1312 return 1;
1313 else if (n_ == header[o_]) // E --> B, B --> C, C --> B
1314 {
1315 o_ = n_;

1317 }
1318 else if (n_ == 'E' and initialization_ == 0) // End of initialization, first EHDR
1319 {
1320 cout << "Initialization done --> EHDR next" << endl;
1321 initialization_ = 1;
1322 return 0;
1323 }
