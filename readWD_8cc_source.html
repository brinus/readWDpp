286{

445DAQEvent &DAQEvent::EvalPedestal()
446{
447 if (!is_getch_)
448 {
449 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
450 exit(0);
451 }
452
453 int ped_interval_dist = ped_interval_.second - ped_interval_.first;
454 const vector<float> &volts = volts_[ch_.first][ch_.second];
455 ped_ = {0., 0.};
456 ped_.first = accumulate(volts.begin() + ped_interval_.first, volts.begin() + ped_interval_.second, 0.) / ped_interval_dist;
457 for (int i = ped_interval_.first; i < ped_interval_.second; ++i)
458 {
459 ped_.second += pow(volts[i] - ped_.first, 2);
460 }
461 ped_.second = sqrt(ped_.second / ped_interval_dist);
462
463 return *this;

472{
473 if (!is_getch_)
474 {
475 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
476 exit(0);

480 {
481 return *this;

484 iw_ = {indexMin_[0], indexMin_[0]};

510DAQEvent &DAQEvent::FindPeaks()
511{
512 if (!is_getch_)
513 {
514 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
515 exit(0);
516 }
517
518 auto &volts = volts_[ch_.first][ch_.second];
519 auto &times = times_[ch_.first][ch_.second];
520 indexMin_ = {};
521
522 if (peak_threshold_ == 1.) // No threshold level set by the user
523 {
524 auto index_min = distance(volts.begin(), min_element(volts.begin(), volts.end()));
525 for (int i = 1; i < SAMPLES_PER_WAVEFORM - 1; ++i)
526 {
527 auto signal = volts[i] - ped_.first < -5 * ped_.second;
528 auto min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
529 auto min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
530 auto at_least = volts[i] < volts[index_min] * 0.5;
531 if (signal and min_left and min_right and at_least)
532 {
533 indexMin_.push_back(i);
534 }
535 }
536
537 if (find(indexMin_.begin(), indexMin_.end(), index_min) == indexMin_.end()) // Sorted insertion of min element's index
538 {
539 auto pos = find_if(indexMin_.begin(), indexMin_.end(), [index_min](auto i)
540 { return i > index_min; });
541 indexMin_.insert(pos, index_min);
542 }
543 }
544 else // Threshold level set by the user
545 {
546 if (user_iw_) // IW set by the user
547 {
548 auto index_min = distance(volts.begin(), min_element(volts.begin() + iw_.first, volts.begin() + iw_.second));
549 indexMin_.push_back(index_min);
550 }
551 else // IW not set by the user
552 {
553 for (int i = 1; i < SAMPLES_PER_WAVEFORM - 1; ++i)
554 {
555 auto signal = volts[i] - ped_.first < -5 * ped_.second;
556 auto min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
557 auto min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
558 auto at_least = volts[i] < peak_threshold_;
559 if (signal and (min_left or min_right) and at_least)
560 {
561 indexMin_.push_back(i);
562 }
563 }
564 }
565 }
566
567 if (indexMin_.size() == 0)
568 {
569 auto index_min = distance(volts.begin(), min_element(volts.begin(), volts.end()));
570 indexMin_.push_back(index_min);
571 }
572
573 peak_ = {volts[indexMin_[0]], times[indexMin_[0]]};
574
575 return *this;
576}
