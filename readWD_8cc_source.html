172void DAQEvent::TimeCalibration(const unsigned short &tCell)
173{
174 float t1, t2, dt;
175
176 // Time sum
177 for (auto &[board_tag, board] : times_)
178 {
179 for (auto &[channel_tag, times] : board)
180 {
181 rotate(times.begin(), times.begin() + tCell, times.end());
182 partial_sum(times.begin(), times.end(), times.begin());
183 }
184 }
185
186 // Time alignement
187 for (auto &[board_tag, board] : times_)
188 {
189 t1 = -1.;
190 for (auto &[channel_tag, channel] : board)
191 {
192 if (t1 == -1.)
193 {
194 t1 = channel[(1024 - tCell) % 1024];
195 }
196 else
197 {
198 t2 = channel[(1024 - tCell) % 1024];
199 dt = t1 - t2;
200 for_each(channel.begin(), channel.end(), [&](float &val)
201 { val += dt; });
202 }
203 }
204 }
205}

362bool DAQFile::operator>>(DRSEvent &event) // DAQFile >> DRSEvent
363{
364 if (!in_.good())
365 {
366 return 0;
367 }
368
369 DAQFile &file = *this;
370 TAG bTag, cTag, tag;
371 EventHeader eh;
372 vector<unsigned short> volts(SAMPLES_PER_WAVEFORM);
373 vector<float> volts_corr(SAMPLES_PER_WAVEFORM);
374
375 // Read only one event
376 file >> eh;
377 if (eh.serialNumber == 1)

383 cout << "Event serial number: " << eh.serialNumber << endl;

385 while (file >> bTag)

391 file >> tag; // Time scaler
392 file.Read(volts);

451DAQFile::operator bool()
452{
453 map<char, char> header{{'E', 'B'}, {'B', 'C'}, {'C', 'B'}};
454 if (!in_.good())
455 {
456 cout << "End of file reached" << endl;
457 return 0;
458 }
459 else if (n_ == 'T' or n_ == 'D') // Ignores DRSx and TIME
460 {
461 return 0;
462 }
463 else if (n_ == o_) // C --> C, B --> B
464 return 1;
465 else if (n_ == header[o_]) // E --> B, B --> C, C --> B
466 {
467 o_ = n_;
468 return 1;
469 }
470 else if (n_ == 'E' and initialization_ == 0) // End of initialization, first EHDR
471 {
472 cout << "Initialization done --> EHDR next" << endl;
473 initialization_ = 1;
474 return 0;
475 }
476 return 0;
