520DAQEvent &DAQEvent::EvalPedestal()
521{
522 if (!is_getch_)
523 {
524 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
525 exit(0);
526 }
527
528 ped_interval_ = config_.pedInterval_[ch_.first][ch_.second];
529 int ped_interval_dist = ped_interval_.second - ped_interval_.first;
530 const vector<float> &volts = volts_[ch_.first][ch_.second];
531 ped_ = {0., 0.};
532 ped_.first = accumulate(volts.begin() + ped_interval_.first, volts.begin() + ped_interval_.second, 0.) / ped_interval_dist;
533 for (int i = ped_interval_.first; i < ped_interval_.second; ++i)
534 {
535 ped_.second += pow(volts[i] - ped_.first, 2);
536 }
537 ped_.second = sqrt(ped_.second / ped_interval_dist);
538
539 return *this;

547DAQEvent &DAQEvent::EvalIntegrationBounds()
548{
549 if (!is_getch_)
550 {
551 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
552 exit(0);
553 }
554
555 if (config_.user_iw_[ch_.first][ch_.second])

560 iw_ = {indexMin_[0], indexMin_[0]};
561
562 const vector<float> &volts = volts_[ch_.first][ch_.second];
563 auto lower_bound = ped_.first - 5 * ped_.second;
564
565 if (peak_.first < lower_bound)
566 {

568 {
569 --iw_.first;

576 }
577

587DAQEvent &DAQEvent::FindPeaks()
588{
589 if (!is_getch_)
590 {
591 cerr << "!! Error: select a channel using DAQEvent::GetChannel()" << endl;
592 exit(0);
593 }
594
595 auto &volts = volts_[ch_.first][ch_.second];
596 auto &times = times_[ch_.first][ch_.second];
597 indexMin_ = {};
598
599 iw_ = config_.intWindow_[ch_.first][ch_.second];
600 peak_threshold_ = config_.peakThr_[ch_.first][ch_.second];
601
602 if (config_.user_iw_[ch_.first][ch_.second]) // Integration window set by the user
603 {
604 auto index_min = distance(volts.begin(), min_element(volts.begin() + iw_.first, volts.begin() + iw_.second));
605 indexMin_.push_back(index_min);
606 }
607 else // No user integration window set
608 {
609 auto index_min = distance(volts.begin() + 2, min_element(volts.begin() + 2, volts.end() - 2)) + 2;
610 for (int i = 2; i < SAMPLES_PER_WAVEFORM - 2; ++i)
611 {
612 auto signal = abs(volts[i] - ped_.first) > 5 * ped_.second;
613 auto min_left = abs(volts[i]) > abs(volts[i - 1]) + ped_.second;
614 auto min_right = abs(volts[i]) > abs(volts[i + 1]) + ped_.second;
615 if (config_.peakThr_[ch_.first][ch_.second] == 0.5) // Default threshold level
616 {
617 auto at_least = abs(volts[i] - ped_.first) > abs(volts[index_min] - ped_.first) * 0.5;
618 }
619 else // Custom threshold level
620 {
621 auto at_least = volts[i] < peak_threshold_;
622 }
623
624 if (signal and min_left and min_right and at_least)
625 {
626 indexMin_.push_back(i);
627 }
628 }
629
630 if (find(indexMin_.begin(), indexMin_.end(), index_min) == indexMin_.end()) // Sorted insertion of min element's index
631 {
632 auto pos = find_if(indexMin_.begin(), indexMin_.end(), [index_min](auto i)
633 { return i > index_min; });
634 indexMin_.insert(pos, index_min);
635 }
636 }
637
638 if (indexMin_.size() == 0) // Assure that at least global minimum is inserted in indexMin_
639 {
640 auto index_min = distance(volts.begin() + 2, min_element(volts.begin() + 2, volts.end() - 2)) + 2;
641 indexMin_.push_back(index_min);
642 }
643
644 peak_ = {volts[indexMin_[0]], times[indexMin_[0]]}; // First local minimum taken as peak
645
646 return *this;
647}

1298DAQFile::operator bool()
1299{
1300 map<char, char> header{{'E', 'B'}, {'B', 'C'}, {'C', 'B'}};
1301 if (!in_.good())
1302 {
1303 cout << "End of file reached" << endl;
1304 return 0;
1305 }
1306 else if (n_ == 'T' or n_ == 'D') // Ignores DRSx and TIME
1307 return 0;
1308 else if (n_ == o_) // C --> C, B --> B
1309 return 1;
1310 else if (n_ == header[o_]) // E --> B, B --> C, C --> B
1311 {
1312 o_ = n_;
1313 return 1;
1314 }
1315 else if (n_ == 'E' and initialization_ == 0) // End of initialization, first EHDR
1316 {
1317 cout << "Initialization done --> EHDR next" << endl;
1318 initialization_ = 1;
1319 return 0;
1320 }
1321 return 0;
